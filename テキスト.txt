// ── WEEK ──────────────────────────────────────────────
function weekKey(d){
  const dt=new Date(d),day=dt.getDay(),diff=dt.getDate()-day+(day===0?-6:1);
  return new Date(new Date(d).setDate(diff)).toISOString().slice(0,10);
}
function fmtWeek(wk){
  const m=new Date(wk),s=new Date(wk);s.setDate(s.getDate()+6);
  const f=d=>`${d.getMonth()+1}/${d.getDate()}`;
  return `${m.getFullYear()}年 ${f(m)}〜${f(s)}`;
}
function initWeek(){
  const inp=document.getElementById('weekIn');
  const today=new Date().toISOString().slice(0,10);
  inp.value=today;week=weekKey(today);
  document.getElementById('weekLbl').textContent=fmtWeek(week);
  inp.addEventListener('change',e=>{
    week=weekKey(e.target.value);
    document.getElementById('weekLbl').textContent=fmtWeek(week);
    render();
  });
}

}

// ── HELPERS ───────────────────────────────────────────
function getTask(sid,tid){
  if(!tasks[week])tasks[week]={};
  if(!tasks[week][sid])tasks[week][sid]={};
  if(!tasks[week][sid][tid])tasks[week][sid][tid]={done:false,note:'',goal:'',familyNote:''};
  return tasks[week][sid][tid];
}
function prog(sid){
  let done=0;
  TASKS.forEach(t=>{if(getTask(sid,t.id).done)done++;});
  return{done,total:TASKS.length,pct:Math.round(done/TASKS.length*100)};
}
